関数呼び出し規約
・関数は何度もあちこちから呼び出される
・呼び出すやつによってレジスタの使い方が違うと困る
（引数が格納されている位置が毎回同じでないと）


let rec gcd2 R1 R2 =
    let m = R1 in
    let n = R2 in
    let zero = 0 in
    if n = zero then m
       	   	else let amari = m mod n in
		     let R1 = n in
		     let R2 = amari in
		     gcd2 R1 R2
